git clone ссылка на репозиторий -- клонирует удалённый репозиторий на комьютер
git status -- показывает информацию о статусе файлов в локальном репозитории
git add filename -- добавление файла в индекс(слежение)
git add . -- добавление файлов в индекс
git commit -m 'Blablabla' -- делает слепок изменения файла для отправки в репозиторий
git commit -a -m / -am 'Blablabla' -- делает слепок изменения всей файлов
git push -- отправляет изменения с локального репозитория в удалённый
git pull -- скачивает изменения с удалённого репозитория в локальный
git fetch -- показывает сравнение локального и удалённого репозитория
git log -- показывает все коммиты которые происходили (q для выхода)
git log --author name_author -- поиск коммита по имени автора
git show хеш коммита -- информация о коммите по его Хешу
git blame filename -- выводит информацию об изменениях авторов в данном файле
git reset HEAD~1 -- откатывает коммит на один шаг назад
git checkout filename -- откатывает изменения в файле
git checkout . -- откатывает изменения во всех файлах
git diff -- показывает изменения которые мы делали в файле
